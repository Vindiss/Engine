315template<typename T>
316std::future<T> Engine::ExecuteFunctionOnThread(std::function <T()> _function)
317{
318 if(actual_nb_threads >= nbCores) {
319 T result = _function();
320 std::promise<T> promise;
321 promise.set_value(result);
322 actual_nb_threads++;
323 return promise.get_future();
324 }
325 else {
326 std::future<T> future = std::async(std::launch::async, _function);
327 actual_nb_threads++;
328
329 return future;
330 }
331}
